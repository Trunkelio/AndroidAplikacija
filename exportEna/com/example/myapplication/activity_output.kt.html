98       // Upload image for health classification
99       private fun uploadHealthImageToApi(imageBitmap: Bitmap) {
100          val file = convertBitmapToFile(imageBitmap)
101          val requestFile = RequestBody.create("multipart/form-data".toMediaTypeOrNull(), file)
102          val body = MultipartBody.Part.createFormData("file", file.name, requestFile)
103  
104          apiService.uploadHealthImage(body).enqueue(object : Callback<HealthResponse> {
105              override fun onResponse(call: Call<HealthResponse>, response: Response<HealthResponse>) {
106                  if (response.isSuccessful) {
107                      val result = response.body()
108                      result?.let {
109                          plantStanjeBox.text = it.health_status  // Display health status
110                      }
111                  } else {
112                      Toast.makeText(this@ActivityOutput, "Failed to get a valid response", Toast.LENGTH_SHORT).show()
113                  }
114              }
115  
116              override fun onFailure(call: Call<HealthResponse>, t: Throwable) {
117                  Toast.makeText(this@ActivityOutput, "Network error: ${t.message}", Toast.LENGTH_SHORT).show()
118              }
119          })
120      }

171      // Helper function to convert Bitmap to file for upload
172      private fun convertBitmapToFile(bitmap: Bitmap): File {
173          val file = File(cacheDir, "temp_image.jpg")
174          val bos = ByteArrayOutputStream()
175          bitmap.compress(Bitmap.CompressFormat.JPEG, 100, bos)
176          file.writeBytes(bos.toByteArray())
177          return file
178      }
